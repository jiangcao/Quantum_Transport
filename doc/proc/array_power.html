    function array_power(A, n) result(C)
        implicit none
        complex(8), intent(in):: A(:, :)
        integer, intent(in) :: n
        complex(8) :: B(size(A, 1), size(A, 1))
        complex(8) :: C(size(A, 1), size(A, 1))
        integer :: ii
        if (n > 0) then
            B = A
            do ii = 2, n
                B = B.m.A
            end do
            C = B
        elseif (n == 0) then
            C = array_eye(size(A, dim=1))
        elseif (n == -1) then
            C = array_inverse(A)
        else
            C = array_inverse(A)
            B = C
            do ii = 2, -n
                B = B.m.C
            end do
            C = B
        end if
    end function array_power
