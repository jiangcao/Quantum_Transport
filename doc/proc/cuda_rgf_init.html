subroutine cuda_rgf_init(nm)
    integer,intent(in)::nm

    Allocate (H_01(nm, nm))
    Allocate (H_10(nm, nm))
    Allocate (Id(nm, nm))     
    Allocate (work(nm, nm))

    Allocate (sig(nm,nm),sigmal(nm,nm),sigmar(nm,nm),sig2(nm,nm),glii(nm,nm),glpii(nm,nm),glnii(nm,nm),cur(nm,nm))

end subroutine cuda_rgf_init
