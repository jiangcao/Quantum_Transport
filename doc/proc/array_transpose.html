    pure function array_transpose(A, t) result(C)
        implicit none
        complex(8), intent(in) :: A(:, :)
        character, intent(in):: t
        complex(8) :: C(size(A, 2), size(A, 1))
        if ((t == 't') .or. (t == 'T')) then
            C = Transpose(A)
        elseif ((t == 'c') .or. (t == 'C')) then
            C = Transpose(Conjg(A))
        end if
    end function array_transpose
